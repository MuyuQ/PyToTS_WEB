## 问题描述

给定两个字符串 `text1` 和 `text2`，返回这两个字符串的最长**公共子序列**的长度。如果不存在公共子序列，返回 `0`。

**子序列**：从原字符串删除或不删除某些字符后，剩余字符保持相对顺序形成的新字符串。

## 思路分析

### 动态规划

- `dp[i][j]`：`text1[0..i-1]` 和 `text2[0..j-1]` 的最长公共子序列长度
- 状态转移：
  - 如果 `text1[i-1] == text2[j-1]`：`dp[i][j] = dp[i-1][j-1] + 1`
  - 否则：`dp[i][j] = max(dp[i-1][j], dp[i][j-1])`

### 空间优化

只需要上一行的状态，可以优化到 O(n) 空间。

## 复杂度分析

- **时间复杂度**：O(m * n)，m和n是两个字符串长度
- **空间复杂度**：O(n)，优化后的一维DP

## Python 实现

```python
def longest_common_subsequence(text1: str, text2: str) -> int:
    """
    最长公共子序列 - 动态规划

    Args:
        text1: 字符串1
        text2: 字符串2

    Returns:
        最长公共子序列长度
    """
    m, n = len(text1), len(text2)

    # dp[i][j] 表示 text1[0..i-1] 和 text2[0..j-1] 的LCS长度
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if text1[i - 1] == text2[j - 1]:
                # 字符相同，LCS长度+1
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                # 字符不同，取较大值
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    return dp[m][n]

def longest_common_subsequence_optimized(text1: str, text2: str) -> int:
    """
    最长公共子序列 - 空间优化

    Args:
        text1: 字符串1
        text2: 字符串2

    Returns:
        最长公共子序列长度
    """
    m, n = len(text1), len(text2)

    # 确保 text2 是较短的，空间复杂度为 O(min(m, n))
    if m < n:
        return longest_common_subsequence_optimized(text2, text1)

    # 只需要两行
    prev = [0] * (n + 1)
    curr = [0] * (n + 1)

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if text1[i - 1] == text2[j - 1]:
                curr[j] = prev[j - 1] + 1
            else:
                curr[j] = max(prev[j], curr[j - 1])
        # 交换
        prev, curr = curr, prev

    return prev[n]

def get_lcs(text1: str, text2: str) -> str:
    """
    获取具体的LCS字符串（不只是长度）

    Args:
        text1: 字符串1
        text2: 字符串2

    Returns:
        最长公共子序列
    """
    m, n = len(text1), len(text2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    # 填充DP表
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if text1[i - 1] == text2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    # 回溯构造LCS
    lcs = []
    i, j = m, n
    while i > 0 and j > 0:
        if text1[i - 1] == text2[j - 1]:
            lcs.append(text1[i - 1])
            i -= 1
            j -= 1
        elif dp[i - 1][j] > dp[i][j - 1]:
            i -= 1
        else:
            j -= 1

    return ''.join(reversed(lcs))


# 测试用例
print(longest_common_subsequence("abcde", "ace"))      # 3 ("ace")
print(longest_common_subsequence("abc", "abc"))        # 3 ("abc")
print(longest_common_subsequence("abc", "def"))        # 0

print(get_lcs("abcde", "ace"))   # "ace"
print(get_lcs("ABAZDC", "BACBAD"))  # "ABAD"
```

## TypeScript 实现

```typescript
/**
 * 最长公共子序列 - 动态规划
 * @param text1 字符串1
 * @param text2 字符串2
 * @returns 最长公共子序列长度
 */
function longestCommonSubsequence(text1: string, text2: string): number {
    const m = text1.length;
    const n = text2.length;

    // dp[i][j] 表示 text1[0..i-1] 和 text2[0..j-1] 的LCS长度
    const dp: number[][] = Array.from({ length: m + 1 }, () =>
        new Array(n + 1).fill(0)
    );

    for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
            if (text1[i - 1] === text2[j - 1]) {
                // 字符相同，LCS长度+1
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                // 字符不同，取较大值
                dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
            }
        }
    }

    return dp[m][n];
}

/**
 * 最长公共子序列 - 空间优化
 * @param text1 字符串1
 * @param text2 字符串2
 * @returns 最长公共子序列长度
 */
function longestCommonSubsequenceOptimized(text1: string, text2: string): number {
    const m = text1.length;
    const n = text2.length;

    // 只需要两行
    let prev: number[] = new Array(n + 1).fill(0);
    let curr: number[] = new Array(n + 1).fill(0);

    for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
            if (text1[i - 1] === text2[j - 1]) {
                curr[j] = prev[j - 1] + 1;
            } else {
                curr[j] = Math.max(prev[j], curr[j - 1]);
            }
        }
        // 交换
        [prev, curr] = [curr, prev];
    }

    return prev[n];
}

/**
 * 获取具体的LCS字符串
 * @param text1 字符串1
 * @param text2 字符串2
 * @returns 最长公共子序列
 */
function getLCS(text1: string, text2: string): string {
    const m = text1.length;
    const n = text2.length;
    const dp: number[][] = Array.from({ length: m + 1 }, () =>
        new Array(n + 1).fill(0)
    );

    // 填充DP表
    for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
            if (text1[i - 1] === text2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
            }
        }
    }

    // 回溯构造LCS
    const lcs: string[] = [];
    let i = m, j = n;
    while (i > 0 && j > 0) {
        if (text1[i - 1] === text2[j - 1]) {
            lcs.push(text1[i - 1]);
            i--;
            j--;
        } else if (dp[i - 1][j] > dp[i][j - 1]) {
            i--;
        } else {
            j--;
        }
    }

    return lcs.reverse().join('');
}

console.log(longestCommonSubsequence("abc", "abc"));        // 3
console.log(longestCommonSubsequence("abc", "def"));        // 0

console.log(getLCS("abcde", "ace"));   // "ace"
console.log(getLCS("ABAZDC", "BACBAD"));  // "ABAD"
```

## 面试变体

1. **最长公共子串**：要求连续，状态转移不同
2. **最长回文子序列**：LCS(s, reverse(s))
3. **编辑距离**：允许插入、删除、替换
4. **最短公共超序列**：包含两个串的最短串
5. **多字符串LCS**：三个或更多字符串的LCS（NP难）

## 常见错误

1. 混淆子序列和子串（连续 vs 不连续）
2. DP数组索引越界（注意i-1和j-1）
3. 空间优化时忘记交换数组
4. 回溯时方向判断错误
