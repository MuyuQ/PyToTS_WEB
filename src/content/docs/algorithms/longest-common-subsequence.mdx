## 问题描述

给定两个字符串 `text1` 和 `text2`，返回这两个字符串的最长**公共子序列**的长度。如果不存在公共子序列，返回 `0`。

**子序列**：从原字符串删除或不删除某些字符后，剩余字符保持相对顺序形成的新字符串。

## 思路分析

### 动态规划

- `dp[i][j]`：`text1[0..i-1]` 和 `text2[0..j-1]` 的最长公共子序列长度
- 状态转移：
  - 如果 `text1[i-1] == text2[j-1]`：`dp[i][j] = dp[i-1][j-1] + 1`
  - 否则：`dp[i][j] = max(dp[i-1][j], dp[i][j-1])`

### 空间优化

只需要上一行的状态，可以优化到 O(n) 空间。

## 复杂度分析

- **时间复杂度**：O(m \* n)，m和n是两个字符串长度
- **空间复杂度**：O(n)，优化后的一维DP

## Python 实现

```python
# 最长公共子序列 - 动态规划 (二维DP + 空间优化 + 回溯构造)
# 算法: Longest Common Subsequence (LCS)
# 时间复杂度: O(m * n), m和n是两个字符串长度
# 空间复杂度: O(m * n) 或 O(n) 优化后

def longest_common_subsequence(text1: str, text2: str) -> int: # 获取两个字符串的长度
m, n = len(text1), len(text2)

    # 创建二维DP表，大小为 (m+1) x (n+1)
    # dp[i][j] 表示 text1[0..i-1] 和 text2[0..j-1] 的LCS长度
    # 初始化为0，表示空字符串的LCS长度为0
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    # 外层循环：遍历 text1 的每个字符（索引1到m）
    for i in range(1, m + 1):
        # 内层循环：遍历 text2 的每个字符（索引1到n）
        for j in range(1, n + 1):
            # 比较当前字符是否相同（注意索引要减1，因为dp表多了一行一列）
            if text1[i - 1] == text2[j - 1]:
                # 字符相同：LCS长度 = 左上角的值 + 1
                # 表示在之前的LCS基础上加上这个相同字符
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                # 字符不同：取上方或左方的较大值
                # dp[i-1][j]：不选 text1[i-1] 的LCS长度
                # dp[i][j-1]：不选 text2[j-1] 的LCS长度
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    # 返回右下角的值，即完整两个字符串的LCS长度
    return dp[m][n]

def longest_common_subsequence_optimized(text1: str, text2: str) -> int: # 获取两个字符串的长度
m, n = len(text1), len(text2)

    # 空间优化：确保 text2 是较短的字符串
    # 这样空间复杂度为 O(min(m, n))
    if m < n:
        return longest_common_subsequence_optimized(text2, text1)

    # 只需要两行：prev 保存上一行状态，curr 计算当前行状态
    prev = [0] * (n + 1)  # 上一行的DP值
    curr = [0] * (n + 1)  # 当前行的DP值

    # 遍历 text1 的每个字符
    for i in range(1, m + 1):
        # 遍历 text2 的每个字符
        for j in range(1, n + 1):
            if text1[i - 1] == text2[j - 1]:
                # 字符相同：LCS长度 = 左上角(prev[j-1]) + 1
                curr[j] = prev[j - 1] + 1
            else:
                # 字符不同：取上方(prev[j])或左方(curr[j-1])的较大值
                curr[j] = max(prev[j], curr[j - 1])

        # 交换两行：当前行变成下一行的"上一行"
        prev, curr = curr, prev

    # 返回最后一行的最后一个值
    return prev[n]

def get_lcs(text1: str, text2: str) -> str: # 获取两个字符串的长度
m, n = len(text1), len(text2)

    # 创建二维DP表
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    # 填充DP表（同上）
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if text1[i - 1] == text2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])

    # 回溯构造具体的LCS字符串
    lcs = []  # 存储LCS字符
    i, j = m, n  # 从DP表右下角开始回溯

    # 从后向前回溯，直到到达边界
    while i > 0 and j > 0:
        if text1[i - 1] == text2[j - 1]:
            # 字符相同：该字符属于LCS，加入结果
            lcs.append(text1[i - 1])
            # 向左上角移动
            i -= 1
            j -= 1
        elif dp[i - 1][j] > dp[i][j - 1]:
            # 上方值更大：说明LCS来自上方，向上移动
            i -= 1
        else:
            # 左方值更大或相等：说明LCS来自左方，向左移动
            j -= 1

    # 回溯得到的是逆序的LCS，需要反转后连接成字符串
    return ''.join(reversed(lcs))

# 测试用例

print(longest_common_subsequence("abcde", "ace")) # 3 ("ace")
print(longest_common_subsequence("abc", "abc")) # 3 ("abc")
print(longest_common_subsequence("abc", "def")) # 0

print(get_lcs("abcde", "ace")) # "ace"
print(get_lcs("ABAZDC", "BACBAD")) # "ABAD"
```

// 最长公共子序列 - 动态规划 (二维DP + 空间优化 + 回溯构造)
// 算法: Longest Common Subsequence (LCS)
// 时间复杂度: O(m _ n), m和n是两个字符串长度
// 空间复杂度: O(m _ n) 或 O(n) 优化后

function longestCommonSubsequence(text1: string, text2: string): number {
// 获取两个字符串的长度
const m = text1.length;
const n = text2.length;

    // 创建二维DP表，大小为 (m+1) x (n+1)
    // dp[i][j] 表示 text1[0..i-1] 和 text2[0..j-1] 的LCS长度
    // 初始化为0，表示空字符串的LCS长度为0
    const dp: number[][] = Array.from({ length: m + 1 }, () =>
        new Array(n + 1).fill(0)
    );

    // 外层循环：遍历 text1 的每个字符（索引1到m）
    for (let i = 1; i <= m; i++) {
        // 内层循环：遍历 text2 的每个字符（索引1到n）
        for (let j = 1; j <= n; j++) {
            // 比较当前字符是否相同（注意索引要减1，因为dp表多了一行一列）
            if (text1[i - 1] === text2[j - 1]) {
                // 字符相同：LCS长度 = 左上角的值 + 1
                // 表示在之前的LCS基础上加上这个相同字符
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                // 字符不同：取上方或左方的较大值
                // dp[i-1][j]：不选 text1[i-1] 的LCS长度
                // dp[i][j-1]：不选 text2[j-1] 的LCS长度
                dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
            }
        }
    }

    // 返回右下角的值，即完整两个字符串的LCS长度
    return dp[m][n];

}

function getLCS(text1: string, text2: string): string {

const m = text1.length;
const n = text2.length;

    // 创建二维DP表
    const dp: number[][] = Array.from({ length: m + 1 }, () =>
        new Array(n + 1).fill(0)
    );

    // 填充DP表（同上）
    for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
            if (text1[i - 1] === text2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
            }
        }

    // 回溯构造具体的LCS字符串
    const lcs: string[] = [];  // 存储LCS字符
    let i = m, j = n;  // 从DP表右下角开始回溯

    // 从后向前回溯，直到到达边界
    while (i > 0 && j > 0) {
        if (text1[i - 1] === text2[j - 1]) {
            // 字符相同：该字符属于LCS，加入结果
            lcs.push(text1[i - 1]);
            // 向左上角移动
            i--;
            j--;

            // 上方值更大：说明LCS来自上方，向上移动
            i--;

            // 左方值更大或相等：说明LCS来自左方，向左移动
            j--;
        }

    return lcs.reverse().join('');

}

## 面试变体

1. **最长公共子串**：要求连续，状态转移不同
2. **最长回文子序列**：LCS(s, reverse(s))
3. **编辑距离**：允许插入、删除、替换
4. **最短公共超序列**：包含两个串的最短串
5. **多字符串LCS**：三个或更多字符串的LCS（NP难）

## 常见错误

1. 混淆子序列和子串（连续 vs 不连续）
2. DP数组索引越界（注意i-1和j-1）
3. 空间优化时忘记交换数组
4. 回溯时方向判断错误
