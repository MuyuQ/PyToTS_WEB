## 问题描述

给你一个链表数组，每个链表都已经按升序排列。

请你将所有链表合并到一个升序链表中，返回合并后的链表。

## 思路分析

### 方法一：最小堆（优先队列）

- 维护一个大小为k的最小堆
- 每次取出堆顶（最小元素），加入该元素的下一个节点
- 时间复杂度：O(N log k)，N是所有节点总数

### 方法二：分治法

- 两两合并，类似归并排序
- 时间复杂度：O(N log k)

### 方法三：顺序合并

- 逐个合并链表
- 时间复杂度：O(N * k)

## 复杂度分析

- **时间复杂度**：O(N log k)，N是所有节点总数，k是链表数
- **空间复杂度**：O(k)，堆的大小

## Python 实现

```python
from typing import Optional, List
import heapq

class ListNode:
    """链表节点"""
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

    # 定义比较方法，用于堆
    def __lt__(self, other):
        return self.val < other.val

def merge_k_lists_heap(lists: List[Optional[ListNode]]) -> Optional[ListNode]:
    """
    合并K个升序链表 - 最小堆法

    Args:
        lists: k个升序链表的头节点列表

    Returns:
        合并后的链表头节点
    """
    # 过滤空链表
    lists = [lst for lst in lists if lst]
    if not lists:
        return None

    # 初始化堆
    heap = []
    for i, node in enumerate(lists):
        # 存储 (值, 索引, 节点)，索引用于打破平局
        heapq.heappush(heap, (node.val, i, node))

    dummy = ListNode(0)
    current = dummy

    while heap:
        val, idx, node = heapq.heappop(heap)
        current.next = node
        current = current.next

        # 将下一个节点加入堆
        if node.next:
            heapq.heappush(heap, (node.next.val, idx, node.next))

    return dummy.next

def merge_k_lists_divide_conquer(lists: List[Optional[ListNode]]) -> Optional[ListNode]]:
    """
    合并K个升序链表 - 分治法

    Args:
        lists: k个升序链表的头节点列表

    Returns:
        合并后的链表头节点
    """
    if not lists:
        return None

    def merge_two_lists(l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        """合并两个有序链表"""
        dummy = ListNode(0)
        current = dummy

        while l1 and l2:
            if l1.val < l2.val:
                current.next = l1
                l1 = l1.next
            else:
                current.next = l2
                l2 = l2.next
            current = current.next

        current.next = l1 if l1 else l2
        return dummy.next

    def merge_range(lists: List[Optional[ListNode]], left: int, right: int) -> Optional[ListNode]:
        """分治合并"""
        if left == right:
            return lists[left]
        if left > right:
            return None

        mid = (left + right) // 2
        l1 = merge_range(lists, left, mid)
        l2 = merge_range(lists, mid + 1, right)
        return merge_two_lists(l1, l2)

    return merge_range(lists, 0, len(lists) - 1)


# 测试用例
def create_list(nums):
    """创建链表"""
    dummy = ListNode(0)
    current = dummy
    for num in nums:
        current.next = ListNode(num)
        current = current.next
    return dummy.next

def print_list(node):
    """打印链表"""
    result = []
    while node:
        result.append(node.val)
        node = node.next
    return result

# 创建3个链表
list1 = create_list([1, 4, 5])
list2 = create_list([1, 3, 4])
list3 = create_list([2, 6])

merged = merge_k_lists_heap([list1, list2, list3])
print(print_list(merged))  # [1, 1, 2, 3, 4, 4, 5, 6]
```

## TypeScript 实现

```typescript
/**
 * 链表节点
 */
class ListNode {
    val: number;
    next: ListNode | null;

    constructor(val?: number, next?: ListNode | null) {
        this.val = val ?? 0;
        this.next = next ?? null;
    }
}

/**
 * 最小堆实现
 */
class MinHeap {
    private data: ListNode[] = [];

    push(node: ListNode): void {
        this.data.push(node);
        this.heapifyUp(this.data.length - 1);
    }

    pop(): ListNode | undefined {
        if (this.data.length === 0) return undefined;
        if (this.data.length === 1) return this.data.pop();

        const root = this.data[0];
        this.data[0] = this.data.pop()!;
        this.heapifyDown(0);
        return root;
    }

    size(): number {
        return this.data.length;
    }

    private heapifyUp(index: number): void {
        while (index > 0) {
            const parent = Math.floor((index - 1) / 2);
            if (this.data[parent].val <= this.data[index].val) break;
            [this.data[parent], this.data[index]] = [this.data[index], this.data[parent]];
            index = parent;
        }
    }

    private heapifyDown(index: number): void {
        while (true) {
            let smallest = index;
            const left = 2 * index + 1;
            const right = 2 * index + 2;

            if (left < this.data.length && this.data[left].val < this.data[smallest].val) {
                smallest = left;
            }
            if (right < this.data.length && this.data[right].val < this.data[smallest].val) {
                smallest = right;
            }

            if (smallest === index) break;
            [this.data[index], this.data[smallest]] = [this.data[smallest], this.data[index]];
            index = smallest;
        }
    }
}

/**
 * 合并K个升序链表 - 最小堆法
 * @param lists k个升序链表的头节点列表
 * @returns 合并后的链表头节点
 */
function mergeKListsHeap(lists: Array<ListNode | null>): ListNode | null {
    // 过滤空链表
    const validLists = lists.filter(lst => lst !== null) as ListNode[];
    if (validLists.length === 0) return null;

    // 初始化堆
    const heap = new MinHeap();
    for (const node of validLists) {
        heap.push(node);
    }

    const dummy = new ListNode(0);
    let current: ListNode = dummy;

    while (heap.size() > 0) {
        const node = heap.pop()!;
        current.next = node;
        current = current.next;

        // 将下一个节点加入堆
        if (node.next) {
            heap.push(node.next);
        }
    }

    return dummy.next;
}

/**
 * 合并两个有序链表
 */
function mergeTwoLists(l1: ListNode | null, l2: ListNode | null): ListNode | null {
    const dummy = new ListNode(0);
    let current: ListNode = dummy;

    while (l1 && l2) {
        if (l1.val < l2.val) {
            current.next = l1;
            l1 = l1.next;
        } else {
            current.next = l2;
            l2 = l2.next;
        }
        current = current.next;
    }

    current.next = l1 ?? l2;
    return dummy.next;
}

/**
 * 合并K个升序链表 - 分治法
 * @param lists k个升序链表的头节点列表
 * @returns 合并后的链表头节点
 */
function mergeKListsDivideConquer(lists: Array<ListNode | null>): ListNode | null {
    if (lists.length === 0) return null;

    function mergeRange(left: number, right: number): ListNode | null {
        if (left === right) return lists[left];
        if (left > right) return null;

        const mid = Math.floor((left + right) / 2);
        const l1 = mergeRange(left, mid);
        const l2 = mergeRange(mid + 1, right);
        return mergeTwoLists(l1, l2);
    }

    return mergeRange(0, lists.length - 1);
}

// 测试用例
function createList(nums: number[]): ListNode | null {
    const dummy = new ListNode(0);
    let current: ListNode = dummy;
    for (const num of nums) {
        current.next = new ListNode(num);
        current = current.next;
    }
    return dummy.next;
}

function printList(node: ListNode | null): number[] {
    const result: number[] = [];
    while (node) {
        result.push(node.val);
        node = node.next;
    }
    return result;
}

// 创建3个链表
const list1 = createList([1, 4, 5]);
const list2 = createList([1, 3, 4]);
const list3 = createList([2, 6]);

const merged = mergeKListsHeap([list1, list2, list3]);
console.log(printList(merged));  // [1, 1, 2, 3, 4, 4, 5, 6]
```

## 面试变体

1. **合并K个数组**：类似思路，但数组支持随机访问
2. **第K小的元素**：在多个有序数组中找第K小
3. **数据流的中位数**：两个堆维护中位数
4. **Top K Frequent**：统计频率最高的K个元素
5. **滑动窗口最大值**：双端队列解法

## 常见错误

1. 忘记处理空链表
2. 堆比较时只比较值，忽略链表节点的引用
3. 分治时边界条件处理错误
4. 忘记将当前节点的next置空（形成环）
